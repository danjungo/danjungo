active_scaffold_input_for(column, scope = nil)

This method decides which input to use for the given column. It does not do any rendering. It only decides which method is responsible for rendering.

[Source]

    # File vendor/plugins/active_scaffold/lib/helpers/form_column_helpers.rb, line 7
 7:       def active_scaffold_input_for(column, scope = nil)
 8:         name = scope ? "record#{scope}[#{column.name}]" : "record[#{column.name}]"
 9:         options = { :name => name, :class => "#{column.name}-input" }
10: 
11:         # first, check if the dev has created an override for this specific field
12:         if override_form_field?(column)
13:           send(override_form_field(column), @record, options[:name])
14: 
15:         # second, check if the dev has specified a valid form_ui for this column
16:         elsif column.form_ui and override_input?(column.form_ui)
17:           send(override_input(column.form_ui), column, options)
18: 
19:         # fallback: we get to make the decision
20:         else
21:           if column.association
22:             # note: i'm not sure if this branch would ever get used. left to our own devices, we render associations as subforms. only if form_ui == :select do we render selects/checkboxes ... and that routing goes through the active_scaffold_input_select method.
23:             raise "Tell the ActiveScaffold team: I'm a real boy!"
24:           elsif column.virtual?
25:             active_scaffold_input_virtual(column, options)
26: 
27:           else # regular model attribute column
28:             # if we (or someone else) have created a custom render option for the column type, use that
29:             if override_input?(column.column.type)
30:               send(override_input(column.column.type), column, options)
31:             # final ultimate fallback: use rails' generic input method
32:             else
33:               # for textual fields we pass different options
34:               text_types = [:text, :string, :integer, :float, :decimal]
35:               options = active_scaffold_input_text_options(options) if text_types.include?(column.column.type)
36: 
37:               input(:record, column.name, options)
38:             end
39:           end
40:         end
41:       end
