 84:       # model_id is the only absolutely required configuration value. it is also not publicly accessible.
 85:       @model_id = model_id.to_s.pluralize.singularize
 86: 
 87:       # inherit the actions list directly from the global level
 88:       @actions = self.class.actions.clone
 89: 
 90:       # create a new default columns datastructure, since it doesn't make sense before now
 91:       attribute_names = self.model.columns.collect{ |c| c.name.to_sym }.sort_by { |c| c.to_s }
 92:       association_column_names = self.model.reflect_on_all_associations.collect{ |a| a.name.to_sym }.sort_by { |c| c.to_s }
 93:       @columns = ActiveScaffold::DataStructures::Columns.new(self.model, attribute_names + association_column_names)
 94: 
 95:       # and then, let's remove some columns from the inheritable set.
 96:       @columns.exclude(*self.class.ignore_columns)
 97:       @columns.exclude(*@columns.find_all { |c| c.column and (c.column.primary or c.column.name =~ /(_id|_count)$/) }.collect {|c| c.name})
 98:       @columns.exclude(*self.model.reflect_on_all_associations.collect{|a| "#{a.name}_type""#{a.name}_type" if a.options[:polymorphic]}.compact)
 99: 
100:       # inherit the global frontend
101:       @frontend = self.class.frontend
102:       @theme = self.class.theme
103: 
104:       # inherit from the global set of action links
105:       @action_links = self.class.action_links.clone
106: 
107:       # the default label
108:       @label = self.model_id.pluralize.titleize
109:     end
