  def test_youtube_attr
    str = "<html><body>\nLorem ipsum. Jolly roger, ding-dong sing-a-long\n<object width=\"425\" height=\"350\">\n<param name=\"movie\" value=\"http://www.youtube.com/v/NbDQ4M_cuwA\"></param>\n<param name=\"wmode\" value=\"transparent\"></param>\n<embed src=\"http://www.youtube.com/v/NbDQ4M_cuwA\"\ntype=\"application/x-shockwave-flash\" wmode=\"transparent\" width=\"425\" height=\"350\">\n</embed>\n</object>\nCheck out my posting, I have bright mice in large clown cars.\n<object width=\"425\" height=\"350\">\n<param name=\"movie\" value=\"http://www.youtube.com/v/foobar\"></param>\n<param name=\"wmode\" value=\"transparent\"></param>\n<embed src=\"http://www.youtube.com/v/foobar\"\ntype=\"application/x-shockwave-flash\" wmode=\"transparent\" width=\"425\" height=\"350\">\n</embed>\n</object>\n</body></html?\n"
    doc = Hpricot(str)
    assert_equal "http://www.youtube.com/v/NbDQ4M_cuwA",
      doc.at("//object/param[@value='http://www.youtube.com/v/NbDQ4M_cuwA']")['value']
  end
